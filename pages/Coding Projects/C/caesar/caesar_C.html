bool isProperString(string input)
{
    for(int i = 0 ; i < strlen(input); i++)
    {
        char curr = input[i];
        if  (!( (curr >= 48 ) && (curr<= 57) )) return false;
    }
    return true;
}

string cipherText(string input , int key){
    string cipher = input;
    for(int i = 0, n = strlen(input) ; i < n ; i++)
    {
        cipher[i] = input[i] + key; 
    }
    return cipher;

}
